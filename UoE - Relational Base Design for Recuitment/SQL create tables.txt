CREATE TABLE EMPLOYEES (
	Employee_ID INTEGER AUTOINCREMENT,
	Role REFERENCES ROLES(Role_ID),
	Turnover_Intention INTEGER,
	Gender TEXT(1),
	Salary INTEGER
);

CREATE TABLE ROLES(
	Role_ID INTEGER,
	Attrition INTEGER,
	Role_Name TEXT(30)
);

CREATE TABLE TESTS(
	Role_ID REFERENCES ROLES(Role_ID),
	Test_Used INTEGER(30)
);


CREATE TABLE CANDIDATES(
	Applicant_ID INTEGER AUTOINCREMENT,
	Name TEXT(30),
	Role REFERENCES ROLES(Role_ID),
	Gender TEXT(1)
);
	
CREATE TABLE PSYCHOMETRIC(
	Applicant_ID REFERENCES CANDIDATES(Applicant_ID),
	Openness INTEGER,
	Neuroticism INTEGER,
	Conscientiousness INTEGER,
	Agreeableness INTEGER,
	Extraversion INTEGER
);

CREATE TABLE NUMERICAL_RATIONING(
	Applicant_ID REFERENCES CANDIDATES(Applicant_ID),
	Score INTEGER
);

CREATE TABLE INTERVIEW(
	Applicant_ID REFERENCES CANDIDATES(Applicant_ID),
	Review TEXT(3000)
);

CREATE TABLE COMPOSITE TEST(
	Applicant_ID REFERENCES CANDIDATES(Applicant_ID),
	Recommend_Role TEXT(60)
);
